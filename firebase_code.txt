  async function syncMembersToFirebase() {
    await db.ref('members').set(S.members || []);
    toast('ซิงก์สมาชิกไป Realtime DB เรียบร้อย', '✅');
  }

  async function loadMembersFromFirebase() {
    const snapshot = await db.ref('members').once('value');
    const data = snapshot.val();
    if (!data) {
      toast('ยังไม่มีสมาชิกใน RTDB', '⚠️');
      return;
    }
    if (Array.isArray(data)) {
      S.members = data;
    } else {
      S.members = Object.values(data);
    }
    save();
    renderAll();
    toast('โหลดสมาชิกจาก Realtime DB เรียบร้อย', '✅');
  }

  window.syncMembersToFirebase = syncMembersToFirebase;
  window.loadMembersFromFirebase = loadMembersFromFirebase;
